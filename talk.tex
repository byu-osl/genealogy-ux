\separator{User Experience}

\frame[plain]{
  \frametitle{User 1}
  \begin{itemize}
  \item distrusts FamilySearch: {\em collaboration difficulties}
    \begin{itemize}
    \item edit wars
    \item little faith in system where there is not consistent sourcing or
      recognition of expertise
    \item angry emails from relatives
    \end{itemize}
  \item prefers desktop to web: {\em control}
  \item topics
    \begin{itemize}
    \item would like research assistance that goes beyond the novice (e.g. Alabama state archives has a collection that you haven't searched)
    \item prefers research log in a separate spreadsheet
    \item collaboration via blogging and manual entry (e.g. copy from blog)
    \item not concerned with analytics
    \end{itemize}
  \item would like
    \begin{itemize}
    \item integration with FamilySearch Wiki and maps
    \end{itemize}
  \end{itemize}
}

\frame[plain]{
  \frametitle{User 2}
  \begin{itemize}
  \item wants everything automated as much as possible: {\em make it easy}
    \begin{itemize}
    \item click on a candidate record, automatically extract
      information and add it to your tree, with sourcing, if it is a
      match
    \item when scanning microfilms, transcribe and upload the record
      to your tree
    \end{itemize}
  \item prefers web to desktop {\em accessible}
  \item topics
    \begin{itemize}
    \item wants research assistance to point out things that need to be done
    \item wants research log online and integrated into family tree
    \item collaboration via a shared tree
      \begin{itemize}
      \item show your family members what you have done lately
      \item assign tasks among family members
      \item resolve conficts via editing hierarchy, voting/reputation,
        employee as a last resort
      \end{itemize}
    \item doesn't care about analytics for bragging rights, but just to be able to remember what you were doing last time
    \end{itemize}
  \end{itemize}
}

\frame[plain]{
  \frametitle{The Present}
  \begin{itemize}
  \item experienced demographic wants
    \begin{itemize}
    \item desktop, control
    \item work individually
    \item excellent online research tools
    \end{itemize}
  \item high quality
    \begin{itemize}
    \item expect genealogy to be done in an exacting way
    \item don't trust online, collaborative environments
    \end{itemize}
  \item {\em they are somewhat jaded}
  \end{itemize}
}

\frame[plain]{
  \frametitle{The Future}
  \begin{itemize}
  \item younger demographic wants
    \begin{itemize}
    \item online, mobile, cloud
    \item highly collaborative
    \item automatic where possible
    \item highly integrated
    \item well-designed tools
    \end{itemize}
  \item high expectations
    \begin{itemize}
    \item add a new document and it populates research log and records
      events and sources on a person all at once, with links between
      the two
    \end{itemize}
  \item {\em they are not yet jaded}
  \end{itemize}
}

\frame[plain]{
  \frametitle{If I Was in Charge}
  \begin{itemize}
  \item product strategy
    \begin{itemize}
    \item online, shared tree
    \item web sites that keep family tree separate but sync with shared tree
    \item mobile products that cache data locally and sync with online
      storage
    \end{itemize}
  \item focus new development on supporting the entire research process
    \begin{itemize}
    \item integrated research logs
    \item integrated research assistants
    \item collaboration tools that leverage reputation and expertise to
      resolve conflicts and end edit wars
    \item collaboration tools that inform family and friends what you
      have been doing and encourage them to help on focused tasks
    \item help users recover what they were doing when they are away for
      a period of time
    \end{itemize}
  \end{itemize}
}
